function loginUser() { let user = document.getElementById("username").value.trim(); let pass = document.getElementById("password").value.trim(); // If fields empty, just go to index.html if(user === "" || pass === ""){ alert("Skipping login, going to shop..."); window.location.href = "index.html"; return; } if(/[^a-zA-Z0-9]/.test(user)){ alert("Invalid characters in username!"); return; } localStorage.setItem("loggedInUser", user); window.location.href = "index.html"; }
